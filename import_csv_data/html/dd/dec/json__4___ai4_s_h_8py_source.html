32class json_db:
33 """
34 @class json_db
35 @brief Handles the processing and structuring of CSV data records for AI4SH in-situ data management, exporting them to hierarchical JSON format.
36
37 @details
38 The json_db class provides methods to:
39 - Initialize with process parameters.
40 - Set up destination folders for output.
41 - Clean and normalize column headers.
42 - Distill and validate parameter, method, equipment, and unit dictionaries.
43 - Check consistency among method, equipment, and unit mappings.
44 - Convert row data to dictionaries using column headers.
45 - Add compulsory parameters with default values if missing.
46 - Ensure all compulsory record parameters are present.
47 - Map equipment to methods and store observations.
48 - Extract and process observation measurements from data rows.
49 - Generate unique identifiers for pilot, site, point, and sample.
50 - Set metadata for observations, samples, and sampling logs.
51 - Assemble hierarchical sample event dictionaries for JSON export.
52 - Write sample event data to JSON files.
53 """

88 def _Set_column_L(self, column_L):
89 """
90 @brief Cleans and normalizes a list of column headers.
91
92 This function processes the input list of column headers by:
93 - Removing any leading Byte Order Mark (BOM) characters and whitespace.
94 - Converting all column names to lowercase.
95 - Replacing values that indicate missing data (e.g., 'na', 'none', 'n/a', 'nan', '', 'null') with None.
96
97 @param column_L List of column header strings to be cleaned and normalized.
98
99 @return None. The cleaned list is assigned to self.column_L.
100 """
101 self.column_L = [col.replace('\ufeff','').strip().lower() for col in column_L]
102 self.column_L = [None if col in ['na', 'none', 'n/a', 'nan', '', 'null'] else col for col in self.column_L]

104 def _Distill_parameters(self, dict_D, parameter_id):
105 """
106 @brief Distills relevant parameters from a dictionary based on the column headers and assigns them to the appropriate attribute.
107
108 This function checks each item in self.column_L against the provided dictionary dict_D. If the item exists and its value is not 'none',
109 it is added to a distilled dictionary. If an item is missing, an error message is printed and None is returned. The distilled dictionary
110 is then assigned to the corresponding attribute (parameter_D, method_D, equipment_D, unit_D, or url_D) based on the parameter_id argument.
111
112 @param dict_D Dictionary containing parameter values to be distilled.
113 @param parameter_id String specifying which attribute to assign the distilled dictionary to. Accepted values: 'parameter', 'method', 'equipment', 'unit', 'url'.
114
115 @return True if successful, None if a required parameter is missing.
116 """
117 distill_D = {}
118
119 for item in self.column_L:
120 if item in dict_D:
121 if dict_D[item] != 'none':
122 distill_D[item] = dict_D[item]
123 else:
124 print(' ❌ ERROR - parameter <%s> is missing in the header dictionary' %(item))
125 print(' Check the header definition file: %s' %(self.process_parameters.method_src_FPN))
126 print(' and/or the data file: %s' %(self.process_parameters.data_src_FPN))
127 return None
128
129 if parameter_id == 'parameter':
130 self.parameter_D = distill_D
131 elif parameter_id == 'method':
132 self.method_D = distill_D
133 elif parameter_id == 'equipment':
134 self.equipment_D = distill_D
135 elif parameter_id == 'unit':
136 self.unit_D = distill_D
137 elif parameter_id == 'url':
138 self.url_D = distill_D
139
140 return True

142 def _Check_parameter_consistency(self):
143 """
144 @brief Checks consistency between method, equipment, and unit parameter dictionaries.
145
146 This function verifies that:
147 - Every key in method_D exists in equipment_D.
148 - Every key in equipment_D exists in unit_D.
149 - Every key in unit_D exists in method_D.
150
151 If any inconsistency is found, an error message is printed and False is returned.
152
153 @return True if all dictionaries are consistent, False otherwise.
154 """
155 for key in self.method_D:
156 if key not in self.equipment_D:
157 print(f" ❌ ERROR - equipment <{key}> is missing in the parameter dictionary")
158 return False
159
160 for key in self.equipment_D:
161 if key not in self.unit_D:
162 print(f" ❌ ERROR - unit <{key}> is missing in the parameter dictionary")
163 return False
164
165 for key in self.unit_D:
166 if key not in self.method_D:
167 print(f" ❌ ERROR - method <{key}> is missing in the parameter dictionary")
168 return False
169
170 return True

250 def _Set_equipment_method(self):
251 """
252 @brief Initializes the equipment-method mapping dictionary.
253
254 @details
255 - Creates a dictionary (equipment_method_D) where each unique equipment is a key, and its value is a dictionary of methods associated with that equipment.
256 - For each method in method_D, associates it with the corresponding equipment in equipment_D and initializes an empty list for storing observations.

265 def _Get_observation_measurements(self, data_row, sd_column = None):
266 """
267 @brief Extracts observation measurements from a data row and appends them to the equipment-method mapping.
268
269 This function iterates over the columns in the data row, identifies measurement columns based on the method dictionary,
270 and processes their values. It handles missing values, values with '<' (interpreted as half the threshold), and optionally
271 includes standard deviation if provided. The processed observation is appended to the corresponding equipment-method list.
272
273 @param data_row List of values representing a single row of measurement data.
274 @param sd_column (Optional) Index of the column containing standard deviation values. If provided, standard deviation is included in the observation dictionary.
275
276 @details
277 - For each measurement column:
278 - If the value is empty, it is set to None.
279 - If the value starts with '<', it is converted to half the threshold value.
280 - The value and standard deviation (if available) are converted to float, with ',' replaced by '.'.
281 - The observation dictionary includes value, unit name, indicator name, lab analysis method name, and optionally standard deviation.
282 - The observation is appended to the equipment-method mapping dictionary.
283
284 @return None
285 """
286 for m, key in enumerate(self.column_L):
287
288 if key in self.method_D:
289
290 if len(data_row[m]) == 0:
291
292 data_row[m] = None
293
294 elif data_row[m][0] == '<':
295
296 data_row[m] = str(float(data_row[m][1:])/2)
297
298 if sd_column:
299
300 observation_D = {'value': None if data_row[m] == 'None' else float(data_row[m].replace(',','.')),
301 'standard_deviation': None if data_row[sd_column] == 'None' else float(data_row[sd_column].replace(',','.')),
302 'unit__name': self.unit_D[key],
303 'indicator__name': self.parameter_D[key],
304 'lab_analysis_method__name': self.method_D[key]}
305
306 else:
307
308 observation_D = {'value': None if data_row[m] == 'None' else float(data_row[m].replace(',','.')),
309 'unit__name': self.unit_D[key],
310 'indicator__name': self.parameter_D[key],
311 'lab_analysis_method__name': self.method_D[key]}
312
313 self.equipment_method_D[self.equipment_D[key]][self.method_D[key]].append(observation_D)

438 def _Assemble_sample_event(self):
439 """
440 @brief Assembles a hierarchical sample event dictionary for JSON export.
441
442 This function constructs a nested dictionary representing a sample event, including data source, site, point, sampling log, sample, observation, and analysis method.
443 The structure is built from previously set metadata and measurement dictionaries within the class instance.
444
445 @details
446 - Combines equipment-method mapping, observation metadata, sample, sampling log, point, site, and data source into a single hierarchical dictionary.
447 - Intended for use in exporting structured sample event data to JSON.
448
449 @return Dictionary containing the complete sample event structure, or None if an error occurs during assembly.
450 """
451
452 try:
453
454 analysis_method = [self.equipment_method_D]
455
456 observation = [{**self.observation_metadata, "analysis_method": analysis_method}]
457
458 sample = [{**self.sample, "observation": observation}]
459
460 sampling_log = [{**self.sampling_log, "sample": sample}]
461
462 point = [{**self.point, "sampling_log": sampling_log}]
463
464 site = [{**self.site, "point": point}]
465
466 data_source = [{**self.data_source, "site": site}]
467
468 sample_event = {"data_source": data_source}
469
470 return sample_event
471
472 except:
473
474 return None

497def Loop_data_records(process, column_L, data_L_L, all_parameter_D, unit_D, method_D, equipment_D, std_row = None):
498 """
499 @brief Processes CSV data records and exports them to hierarchical JSON format for AI4SH in-situ data management.
500
501 @details
502 This function orchestrates the conversion of CSV data rows into structured JSON sample events.
503 It initializes the json_db class, sets up output folders, cleans column headers, distills parameter dictionaries,
504 checks consistency, and loops through each data record to assemble and export sample events.
505
506 - Initializes the json_db class with process parameters.
507 - Sets up the destination folder for JSON output.
508 - Cleans and normalizes column headers.
509 - Distills parameter, method, equipment, and unit dictionaries.
510 - Checks consistency among method, equipment, and unit mappings.
511 - Adds compulsory parameters with default values if missing.
512 - Loops through each data record:
513 - Initializes equipment-method mapping.
514 - Converts row data to dictionary.
515 - Ensures all compulsory parameters are set.
516 - Extracts and processes observation measurements.
517 - Sets unique identifiers for pilot, site, point, and sample.
518 - Sets observation metadata, sample, and sampling log.
519 - Assembles the sample event and exports to JSON.
520
521 @param process Object containing process parameters, expected to have a 'parameters' attribute.
522 @param column_L List of column header strings from the CSV file.
523 @param data_L_L List of lists, each representing a row of data from the CSV file.
524 @param all_parameter_D Dictionary mapping column headers to parameter names.
525 @param unit_D Dictionary mapping column headers to unit names.
526 @param method_D Dictionary mapping column headers to method names.
527 @param equipment_D Dictionary mapping column headers to equipment names.
528 @param std_row (Optional) Standard deviation row or index, if available.
529
530 @return None if any error occurs during processing, otherwise creates JSON files for each sample event.
531 """
532 # Initiate the json_db class
533 json_db_C = json_db(process)
534
535 # Create the destination folder if it doesn't exist
536 json_db_C._Set_dst_FP()
537
538 # Clean and set the column headers
539 json_db_C._Set_column_L(column_L)
540
541 parameter_id_L =['parameter','method','equipment','unit']
542
543 # Distill the parameter, method, equipment and unit dictionaries
544 for i, item in enumerate([all_parameter_D, method_D, equipment_D, unit_D]):
545
546 result = json_db_C._Distill_parameters( item, parameter_id_L[i])
547
548 if not result:
549
550 return None
551
552 # Check that the method, equipment and unit parameters are consistent
553 result = json_db_C._Check_parameter_consistency()
554
555 if not result:
556
557 return None
558
559 # Add compulsary parameters with default values if they are not set in the csv data file
560 json_db_C._Add_compulsary_default_parameters()
561
562 # Loop all the data records in the csv data file
563 for d,data_row in enumerate(data_L_L):
564
565 # Create a hierarchical dictionary to hold equipment -> methods (must be recreated in each loop)
566 json_db_C._Set_equipment_method()
567
568 # Convert the csv data row to a dict using the csv header records as keys
569 json_db_C._Row_data_to_dict(data_row)
570
571 # Check that all compulsory parameters are set in in the row data dictionary
572 result = json_db_C._Check_set_compulsary_record_parameters()
573
574 if not result:
575
576 return None
577
578 # Set the observed measurements linked to the correct equipment and method
579 json_db_C._Get_observation_measurements(data_row)
580
581 # Set the ids for pilot, site, point and sample
582 json_db_C._Set_pilot_site_point_sample_id()
583
584 # Set the observation metadata
585 result = json_db_C._Set_observation_metadata()
586
587 if not result:
588
589 return None
590
591 # Set the sample parameters
592 json_db_C._Set_sample()
593
594 # Set the sampling log parameters
595 json_db_C._Set_sampling_log()
596
597 # Assemble the complete sample event to a final dictionary containing all parameters
598 sample_event = json_db_C._Assemble_sample_event()
599
600 if sample_event:
601
602 # Dump the complete sample event to a JSON file
603 json_db_C._Dump_sample_json(sample_event)
604
605 else:
606
607 print('❌ Error creating JSON post')
608
609 return None

611def Parameters_fix(method_src_FPN):
612 """
613 @brief Extracts parameter, unit, method, and equipment dictionaries from a method definition CSV file.
614
615 @details
616 This function reads a method definition CSV file, processes its contents, and returns four dictionaries mapping headers to their respective parameter, unit, method, and equipment values. It performs the following steps:
617 - Checks if the provided file path exists.
618 - Reads the CSV file and unpacks the header and data rows.
619 - Normalizes header and row values to lowercase and replaces missing values with None.
620 - Constructs column-wise lists for each header.
621 - Creates dictionaries for parameters, units, methods, and equipment using the header as keys.
622
623 @param method_src_FPN Absolute file path to the method definition CSV file.
624
625 @return Tuple containing:
626 - parameter_D (dict): Maps header to parameter name.
627 - unit_D (dict): Maps header to unit name.
628 - method_D (dict): Maps header to method name.
629 - equipment_D (dict): Maps header to equipment name.
630 Returns None if the file does not exist or if an error occurs during processing.
631
632 @exception Prints error messages if the file path does not exist or if dictionary creation fails.
633 """
634
635 data_pack = Data_read(method_src_FPN)
636
637 if not data_pack:
638
639 return None
640
641 # Unpack the CSV to a header column list and a list of lists of data rows
642 column_L, data_L_L = data_pack
643
644 # Convert header column names to lowercase
645 column_L = [col.strip().lower() for col in column_L]
646
647 # Create a dictionary to hold column data
648 column_D = {}
649
650 for item in column_L:
651
652 column_D[item] = []
653
654 # Loop over the data rows and populate the column data dictionary
655 for row in data_L_L:

657 for c, col in enumerate(column_L):

661 if row_item.lower() in ['na', 'none', 'n/a', 'nan', '', 'null']:
662
663 row[c] = None
664
665 column_D[col].append(row_item)

667 # Create a parameter dictionary
668 try:
669
670 parameter_D = dict(zip(column_D['header'], column_D['parameter']))

681
682 # Create an equipment dictionary from the column data
